001package io.react.runtime.testee;
002
003import io.react.Action;
004import io.react.atmosphere.AtmosphereBridge;
005import io.react.runtime.DefaultServer;
006import io.react.runtime.Server;
007import io.react.runtime.Socket;
008import io.react.runtime.Socket.Reply;
009
010import javax.servlet.ServletContextEvent;
011import javax.servlet.ServletContextListener;
012import javax.servlet.annotation.WebListener;
013
014/**
015 * Bootstrap to start the testee server.
016 * <p>
017 * To start the server:
018 * <p>
019 * {@code $ mvn jetty:run -Djetty.port=8000}
020 * <p>
021 * How to run the test suite is available at 
022 * https://github.com/atmosphere/react-protocol#testing
023 * 
024 * @author Donghwan Kim
025 */
026@WebListener
027public class ServerBootstrap implements ServletContextListener {
028
029    @Override
030    public void contextInitialized(ServletContextEvent event) {
031        Server server = new DefaultServer();
032
033        server.socketAction(new Action<Socket>() {
034            @Override
035            public void on(final Socket socket) {
036                socket.on("echo", new Action<Object>() {
037                    @Override
038                    public void on(Object data) {
039                        socket.send("echo", data);
040                    }
041                })
042                .on("reaction", new Action<Reply<Boolean>>() {
043                    @Override
044                    public void on(Reply<Boolean> reply) {
045                        if (reply.data()) {
046                            reply.resolve(reply.data());
047                        } else {
048                            reply.reject(reply.data());
049                        }
050                    }
051                });
052            }
053        });
054
055        new AtmosphereBridge(event.getServletContext(), "/react").httpAction(server.httpAction()).websocketAction(server.websocketAction());
056    }
057
058    @Override
059    public void contextDestroyed(ServletContextEvent sce) {}
060
061}

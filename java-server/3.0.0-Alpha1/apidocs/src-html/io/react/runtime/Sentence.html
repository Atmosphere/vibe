016package io.react.runtime;
017
018import io.react.Action;
019
020import java.io.Serializable;
021
022/**
023 * {@code Sentence} is a series of predicates that a group of socket have to
024 * follow. It makes easy to write one-liner action and uses internally built
025 * actions implementing {@link Serializable} that is typically needed in cluster
026 * environments. Use of {@code Sentence} is preferred to that of action if the
027 * goal is the same.
028 * 
029 * @author Donghwan Kim
030 */
031public class Sentence implements AbstractSocket<Sentence> {
032
033    private final Action<Action<Socket>> serverAction;
034
035    Sentence(Action<Action<Socket>> serverAction) {
036        this.serverAction = serverAction;
037    }
038
039    @Override
040    public Sentence send(String event) {
041        return send(event, null);
042    }
043
044    @Override
045    public Sentence send(String event, Object data) {
046        execute(new SendAction(event, data));
047        return this;
048    }
049
050    @Override
051    public Sentence close() {
052        execute(new CloseAction());
053        return this;
054    }
055
056    @Override
057    public Sentence tag(String... names) {
058        execute(new TagAction(names));
059        return this;
060    }
061
062    @Override
063    public Sentence untag(String... names) {
064        execute(new UntagAction(names));
065        return this;
066    }
067
068    private void execute(Action<Socket> action) {
069        serverAction.on(action);
070    }
071
072    static interface SerializableAction<T> extends Action<T>, Serializable {
073    }
074
075    static class SendAction implements SerializableAction<Socket> {
076        private static final long serialVersionUID = 2178442626501531717L;
077        final String event;
078        final Object data;
079
080        SendAction(String event, Object data) {
081            this.event = event;
082            this.data = data;
083        }
084
085        @Override
086        public void on(Socket socket) {
087            socket.send(event, data);
088        }
089    }
090
091    static class CloseAction implements SerializableAction<Socket> {
092        private static final long serialVersionUID = 8154281469036373698L;
093
094        @Override
095        public void on(Socket socket) {
096            socket.close();
097        }
098    }
099
100    static class TagAction implements SerializableAction<Socket> {
101        private static final long serialVersionUID = -7789207688974771161L;
102        final String[] names;
103
104        public TagAction(String[] names) {
105            this.names = names;
106        }
107
108        @Override
109        public void on(Socket socket) {
110            socket.tag(names);
111        }
112    }
113
114    static class UntagAction implements SerializableAction<Socket> {
115        private static final long serialVersionUID = -4173842573981245930L;
116        final String[] names;
117
118        public UntagAction(String[] names) {
119            this.names = names;
120        }
121
122        @Override
123        public void on(Socket socket) {
124            socket.untag(names);
125        }
126    }
127
128}

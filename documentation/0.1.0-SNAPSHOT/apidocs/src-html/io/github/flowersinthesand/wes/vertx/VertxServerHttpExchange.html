001package io.github.flowersinthesand.wes.vertx;
002
003import io.github.flowersinthesand.wes.Data;
004import io.github.flowersinthesand.wes.http.AbstractServerHttpExchange;
005import io.github.flowersinthesand.wes.http.ServerHttpExchange;
006import io.github.flowersinthesand.wes.http.StatusCode;
007
008import java.util.List;
009import java.util.Set;
010
011import org.vertx.java.core.Handler;
012import org.vertx.java.core.VoidHandler;
013import org.vertx.java.core.buffer.Buffer;
014import org.vertx.java.core.http.HttpServerRequest;
015
016public class VertxServerHttpExchange extends AbstractServerHttpExchange {
017
018    private final HttpServerRequest request;
019
020    public VertxServerHttpExchange(HttpServerRequest request) {
021        this.request = request;
022        request.response().setChunked(true).closeHandler(new VoidHandler() {
023            @Override
024            protected void handle() {
025                closeActions.fire();
026            }
027        });
028    }
029
030    @Override
031    public String uri() {
032        return request.uri();
033    }
034
035    @Override
036    public String method() {
037        return request.method();
038    }
039
040    @Override
041    public Set<String> requestHeaderNames() {
042        return request.headers().names();
043    }
044
045    @Override
046    public List<String> requestHeaders(String name) {
047        return request.headers().getAll(name);
048    }
049
050    @Override
051    protected void readBody() {
052        request.bodyHandler(new Handler<Buffer>() {
053            @Override
054            public void handle(Buffer body) {
055                bodyActions.fire(new Data(body.toString()));
056            }
057        });
058    }
059
060    @Override
061    public ServerHttpExchange setResponseHeader(String name, String value) {
062        request.response().putHeader(name, value);
063        return this;
064    }
065
066    @Override
067    protected void doWrite(String data) {
068        request.response().write(data);
069    }
070
071    @Override
072    protected void doClose() {
073        request.response().end();
074        request.response().close();
075    }
076
077    @Override
078    public ServerHttpExchange setStatus(StatusCode status) {
079        request.response().setStatusCode(status.code()).setStatusMessage(status.reason());
080        return this;
081    }
082
083    @Override
084    public <T> T unwrap(Class<T> clazz) {
085        return HttpServerRequest.class.isAssignableFrom(clazz) ? clazz.cast(request) : null;
086    }
087
088}

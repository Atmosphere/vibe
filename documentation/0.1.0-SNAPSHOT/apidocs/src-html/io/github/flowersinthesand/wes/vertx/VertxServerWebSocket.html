001package io.github.flowersinthesand.wes.vertx;
002
003import io.github.flowersinthesand.wes.Data;
004import io.github.flowersinthesand.wes.websocket.AbstractServerWebSocket;
005
006import org.vertx.java.core.Handler;
007import org.vertx.java.core.VoidHandler;
008import org.vertx.java.core.buffer.Buffer;
009import org.vertx.java.core.http.ServerWebSocket;
010
011public class VertxServerWebSocket extends AbstractServerWebSocket {
012
013    private final org.vertx.java.core.http.ServerWebSocket socket;
014    private String uri;
015
016    public VertxServerWebSocket(org.vertx.java.core.http.ServerWebSocket socket) {
017        this.socket = socket;
018        socket.closeHandler(new VoidHandler() {
019            @Override
020            protected void handle() {
021                closeActions.fire();
022            }
023        })
024        .exceptionHandler(new Handler<Throwable>() {
025            @Override
026            public void handle(Throwable throwable) {
027                errorActions.fire(throwable);
028            }
029        })
030        .dataHandler(new Handler<Buffer>() {
031            @Override
032            public void handle(Buffer buffer) {
033                messageActions.fire(new Data(buffer.toString()));
034            }
035        });
036    }
037
038    @Override
039    public String uri() {
040        if (uri == null) {
041            uri = socket.path();
042            if (socket.query() != null) {
043                uri += "?" + socket.query();
044            }
045        }
046        return uri;
047    }
048
049    @Override
050    protected void doClose() {
051        socket.close();
052    }
053
054    @Override
055    protected void doSend(String data) {
056        socket.writeTextFrame(data);
057    }
058
059    @Override
060    public <T> T unwrap(Class<T> clazz) {
061        return ServerWebSocket.class.isAssignableFrom(clazz) ? clazz.cast(socket) : null;
062    }
063
064}

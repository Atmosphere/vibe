016package org.atmosphere.vibe.servlet;
017
018import org.atmosphere.vibe.AbstractServerHttpExchange;
019import org.atmosphere.vibe.Actions;
020import org.atmosphere.vibe.Data;
021import org.atmosphere.vibe.HttpStatus;
022import org.atmosphere.vibe.ServerHttpExchange;
023
024import javax.servlet.AsyncContext;
025import javax.servlet.AsyncEvent;
026import javax.servlet.AsyncListener;
027import javax.servlet.ReadListener;
028import javax.servlet.ServletInputStream;
029import javax.servlet.http.HttpServletRequest;
030import javax.servlet.http.HttpServletResponse;
031import java.io.IOException;
032import java.io.PrintWriter;
033import java.nio.charset.Charset;
034import java.util.Collections;
035import java.util.Enumeration;
036import java.util.LinkedHashSet;
037import java.util.List;
038import java.util.Set;
039
040/**
041 * {@link ServerHttpExchange} for Servlet 3.
042 *
043 * @author Donghwan Kim
044 */
045public class ServletServerHttpExchange extends AbstractServerHttpExchange {
046
047    private final HttpServletRequest request;
048    private final HttpServletResponse response;
049
050    public ServletServerHttpExchange(HttpServletRequest request, HttpServletResponse response) {
051        this.request = request;
052        this.response = response;
053        AsyncContext async = request.startAsync();
054        async.setTimeout(0);
055        async.addListener(new AsyncListener() {
056            @Override
057            public void onStartAsync(AsyncEvent event) throws IOException {
058            }
059
060            @Override
061            public void onTimeout(AsyncEvent event) throws IOException {
062                closeActions.fire();
063            }
064
065            @Override
066            public void onError(AsyncEvent event) throws IOException {
067                closeActions.fire();
068            }
069
070            @Override
071            public void onComplete(AsyncEvent event) throws IOException {
072                closeActions.fire();
073            }
074        });
075    }
076
077    @Override
078    public String uri() {
079        String uri = request.getRequestURI();
080        if (request.getQueryString() != null) {
081            uri += "?" + request.getQueryString();
082        }
083        return uri;
084    }
085
086    @Override
087    public String method() {
088        return request.getMethod();
089    }
090
091    @Override
092    public Set<String> requestHeaderNames() {
093        Set<String> headerNames = new LinkedHashSet<>();
094        Enumeration<String> enumeration = request.getHeaderNames();
095        while (enumeration.hasMoreElements()) {
096            headerNames.add(enumeration.nextElement());
097        }
098        return headerNames;
099    }
100
101    @Override
102    public List<String> requestHeaders(String name) {
103        return Collections.list(request.getHeaders(name));
104    }
105
106    @Override
107    protected void readBody() {
108        final ServletInputStream input;
109        try {
110            input = request.getInputStream();
111        } catch (IOException e) {
112            throw new RuntimeException();
113        }
114        // HTTP 1.1 says that the default charset is ISO-8859-1
115        // http://www.w3.org/International/O-HTTP-charset#charset
116        String charsetName = request.getCharacterEncoding();
117        final Charset charset = Charset.forName(charsetName == null ? "ISO-8859-1" : charsetName);
118
119        if (request.getServletContext().getMinorVersion() > 0) {
120            // 3.1+ asynchronous
121            new AsyncBodyReader(input, charset, bodyActions);
122        } else {
123            // 3.0 synchronous
124            new SyncBodyReader(input, charset, bodyActions);
125        }
126    }
127
128    private abstract static class BodyReader {
129        final ServletInputStream input;
130        final Charset charset;
131        final Actions<Data> actions;
132        final StringBuilder body = new StringBuilder();
133
134        public BodyReader(ServletInputStream input, Charset charset, Actions<Data> bodyActions) {
135            this.input = input;
136            this.charset = charset;
137            this.actions = bodyActions;
138            start();
139        }
140
141        abstract void start();
142
143        void read() throws IOException {
144            int bytesRead = -1;
145            byte buffer[] = new byte[4096];
146            while (ready() && (bytesRead = input.read(buffer)) != -1) {
147                String data = new String(buffer, 0, bytesRead, charset);
148                body.append(data);
149            }
150        }
151
152        abstract boolean ready();
153
154        void end() {
155            actions.fire(new Data(body.toString()));
156        }
157    }
158
159    private static class AsyncBodyReader extends BodyReader {
160        public AsyncBodyReader(ServletInputStream input, Charset charset, Actions<Data> bodyActions) {
161            super(input, charset, bodyActions);
162        }
163
164        @Override
165        void start() {
166            input.setReadListener(new ReadListener() {
167                @Override
168                public void onDataAvailable() throws IOException {
169                    read();
170                }
171
172                @Override
173                public void onAllDataRead() throws IOException {
174                    end();
175                }
176
177                @Override
178                public void onError(Throwable t) {
179                    throw new RuntimeException(t);
180                }
181            });
182        }
183
184        @Override
185        boolean ready() {
186            return input.isReady();
187        }
188    }
189
190    private static class SyncBodyReader extends BodyReader {
191        public SyncBodyReader(ServletInputStream input, Charset charset, Actions<Data> bodyActions) {
192            super(input, charset, bodyActions);
193        }
194
195        @Override
196        void start() {
197            try {
198                read();
199                end();
200            } catch (IOException e) {
201                throw new RuntimeException(e);
202            }
203        }
204
205        @Override
206        boolean ready() {
207            return true;
208        }
209    }
210
211    @Override
212    protected void doSetResponseHeader(String name, String value) {
213        response.setHeader(name, value);
214    }
215
216    @Override
217    protected void doWrite(byte[] data, int offset, int length) {
218        try {
219            response.getOutputStream().write(data, offset, length);
220        } catch (IOException e) {
221            throw new RuntimeException(e);
222        }
223    }
224
225    @Override
226    protected void doSetStatus(HttpStatus status) {
227        response.setStatus(status.code());
228    }
229
230    @Override
231    protected void doWrite(String data) {
232        try {
233            PrintWriter writer = response.getWriter();
234            writer.print(data);
235            writer.flush();
236        } catch (IOException e) {
237            throw new RuntimeException(e);
238        }
239    }
240
241    @Override
242    protected void doClose() {
243        request.getAsyncContext().complete();
244    }
245
246    /**
247     * {@link HttpServletRequest} and {@link HttpServletResponse} are available.
248     */
249    @Override
250    public <T> T unwrap(Class<T> clazz) {
251        return HttpServletRequest.class.isAssignableFrom(clazz) ?
252                clazz.cast(request) :
253                HttpServletResponse.class.isAssignableFrom(clazz) ?
254                        clazz.cast(response) :
255                        null;
256    }
